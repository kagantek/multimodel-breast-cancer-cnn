class DataModule:
    def __init__(self, cfg: Config):
        self.cfg = cfg
        self.label_map = {"benign": 0, "normal": 0, "malignant": 1}
        self.train_files: List[str] = []
        self.val_files: List[str] = []
        self.test_files: List[str] = []
        self.train_labels: List[int] = []
        self.val_labels: List[int] = []
        self.test_labels: List[int] = []
        self.train_ds = self.val_ds = self.test_ds = None
        random.seed(cfg.SEED)
        np.random.seed(cfg.SEED)

    def prepare_splits(self) -> None:
        paths, labels = self._collect_files()
        self._split(paths, labels)

    def build_tf_datasets(self) -> None:
        self.train_ds = self._make_dataset(self.train_files, self.train_labels, training=True)
        self.val_ds   = self._make_dataset(self.val_files,   self.val_labels,   training=False)
        self.test_ds  = self._make_dataset(self.test_files,  self.test_labels,  training=False)

    def _collect_files(self) -> Tuple[List[str], List[int]]:
        root = Path(self.cfg.ROOT)
        files, labels = [], []
        for cls, lbl in self.label_map.items():
            for path in root.rglob(f"**/{cls}/*.png"):
                files.append(str(path))
                labels.append(lbl)
        if not files:
            raise FileNotFoundError("No PNG files found. Check ROOT path and subfolder names.")

        if self.cfg.LIMIT is not None and self.cfg.POS_RATIO is not None:
            pos_quota = int(self.cfg.LIMIT * self.cfg.POS_RATIO)
            neg_quota = self.cfg.LIMIT - pos_quota
            pos_files = [p for p, l in zip(files, labels) if l == 1]
            neg_files = [p for p, l in zip(files, labels) if l == 0]
            pos_sel = random.sample(pos_files, min(pos_quota, len(pos_files)))
            neg_sel = random.sample(neg_files, min(neg_quota, len(neg_files)))
            files  = pos_sel + neg_sel
            labels = [1] * len(pos_sel) + [0] * len(neg_sel)
        tmp = list(zip(files, labels))
        random.shuffle(tmp)
        return zip(*tmp)  

    def _split(self, files: List[str], labels: List[int]) -> None:
        from sklearn.model_selection import train_test_split
        x_tmp, x_test, y_tmp, y_test = train_test_split(
            files, labels, test_size=0.15, stratify=labels, random_state=self.cfg.SEED)
        val_frac = 0.15 / 0.85
        x_train, x_val, y_train, y_val = train_test_split(
            x_tmp, y_tmp, test_size=val_frac, stratify=y_tmp, random_state=self.cfg.SEED)
        self.train_files, self.val_files, self.test_files = list(x_train), list(x_val), list(x_test)
        self.train_labels, self.val_labels, self.test_labels = list(y_train), list(y_val), list(y_test)

    def _decode(self, path: tf.Tensor, label: tf.Tensor):
        img = tf.io.read_file(path)
        img = tf.io.decode_png(img, channels=3)
        img = tf.image.resize(img, self.cfg.IMG_SIZE)
        return tf.cast(img, tf.float32), label

    def _augment_fn(self):
        return tf.keras.Sequential([
            layers.RandomFlip("horizontal"),
            layers.RandomRotation(self.cfg.AUGMENT_KWARGS["rotation_range"]),
            layers.RandomZoom(self.cfg.AUGMENT_KWARGS["zoom_range"]),
            layers.RandomBrightness(self.cfg.AUGMENT_KWARGS["brightness_range"][1] - 1),
        ])

    def _make_dataset(self, paths: List[str], labels: List[int], training=False):
        ds = tf.data.Dataset.from_tensor_slices((paths, labels))
        if training:
            ds = ds.shuffle(len(paths), seed=self.cfg.SEED, reshuffle_each_iteration=True)
        ds = ds.map(self._decode, num_parallel_calls=tf.data.AUTOTUNE)
        if training:
            aug = self._augment_fn()
            ds = ds.map(lambda x, y: (aug(x, training=True), y), num_parallel_calls=tf.data.AUTOTUNE)
        return ds.batch(self.cfg.BATCH).prefetch(tf.data.AUTOTUNE)

class ModelFactory:
    @staticmethod
    def _build_backbone(backbone_fn, preprocess, cfg: Config, train_tail: List[str]):
        inputs = layers.Input(shape=(*cfg.IMG_SIZE, 3), name="input_rgb")
        x = layers.Lambda(preprocess, name="preproc")(inputs)
        backbone = backbone_fn(include_top=False, weights="imagenet", input_tensor=x)
        for l in backbone.layers:
            l.trainable = False
        x = layers.GlobalAveragePooling2D()(backbone.output)
        x = layers.Dropout(0.5)(x)
        x = layers.Dense(256, activation="relu")(x)
        x = layers.Dropout(0.25)(x)
        out = layers.Dense(1, activation="sigmoid", name="pred")(x)
        model = models.Model(inputs, out)
        model._unfreeze_prefix = train_tail  # type: ignore
        model.compile(optimizer=tf.keras.optimizers.Adam(cfg.LR_STAGE1),
                      loss="binary_crossentropy",
                      metrics=["accuracy", tf.keras.metrics.AUC(name="auc"),
                               tf.keras.metrics.Precision(name="precision"),
                               tf.keras.metrics.Recall(name="recall")])
        return model

    @classmethod
    def build_resnet50(cls, cfg: Config):
        return cls._build_backbone(ResNet50, resnet50.preprocess_input, cfg, ["conv4", "conv5"])

    @classmethod
    def build_vgg16(cls, cfg: Config):
        from tensorflow.keras.applications import VGG16 as _VGG16
        return cls._build_backbone(_VGG16, vgg16.preprocess_input, cfg, ["block5"])

    @classmethod
    def build_densenet121(cls, cfg: Config):
        return cls._build_backbone(DenseNet121, densenet.preprocess_input, cfg, ["conv4", "conv5"])

class Trainer:
    def __init__(self, cfg: Config):
        self.cfg = cfg

    def fine_tune(self, model: models.Model, train_ds, val_ds):
        steps_per_epoch = math.ceil(len(train_ds) * self.cfg.BATCH / self.cfg.BATCH)
        hist1 = model.fit(train_ds,
                          validation_data=val_ds,
                          epochs=self.cfg.EPOCHS_STAGE1,
                          steps_per_epoch=steps_per_epoch,
                          verbose=2)
        for layer in model.layers:
            if any(layer.name.startswith(p) for p in getattr(model, "_unfreeze_prefix", [])):
                layer.trainable = True
        model.compile(optimizer=tf.keras.optimizers.Adam(self.cfg.LR_STAGE2),
                      loss="binary_crossentropy",
                      metrics=model.metrics)
        hist2 = model.fit(train_ds,
                          validation_data=val_ds,
                          epochs=self.cfg.EPOCHS_STAGE2,
                          steps_per_epoch=steps_per_epoch,
                          verbose=2)
        for k in hist1.history:
            hist1.history[k] += hist2.history[k]
        return hist1

class Evaluator:
    def __init__(self, cfg: Config):
        self.cfg = cfg

    def predict_dataset(self, model, ds):
        y_true = np.concatenate([y.numpy() for _, y in ds], axis=0)
        y_prob = model.predict(ds, verbose=0).ravel()
        return y_true, y_prob

    def metrics_dict(self, y_true, y_prob, thr=0.5):
        y_pred = (y_prob >= thr).astype(int)
        return dict(
            accuracy=accuracy_score(y_true, y_pred),
            precision=precision_score(y_true, y_pred, zero_division=0),
            recall=recall_score(y_true, y_pred, zero_division=0),
            f1=f1_score(y_true, y_pred, zero_division=0),
            auc=roc_auc_score(y_true, y_prob),
        )
    
    def search_threshold(self, y_true, y_prob):
        best = {"tau":None, "f1":0, "prec":0, "rec":0}
        for t in np.linspace(0.01,0.99,99):
            y_hat = (y_prob>=t).astype(int)
            p,r,f,_ = precision_score(y_true,y_hat), recall_score(y_true,y_hat), f1_score(y_true,y_hat), None
            if f>best['f1']:
                best.update({"tau":t, "f1":f, "prec":p, "rec":r})
        return best

    def full_report(self, model, test_ds, history):
        y_true, y_prob = self.predict_dataset(model, test_ds)
        stats = self.metrics_dict(y_true, y_prob, thr=0.5)
        print("\nTest metrics:")
        for k, v in stats.items():
            print(f"{k:10s}: {v:.4f}")
        print("\nDetailed report:\n", classification_report(y_true, (y_prob>=0.5).astype(int), digits=4))
        self._plot_learning_curves(history)
        self._plot_confusion(y_true, y_prob)
        self._plot_roc(y_true, y_prob)

    def _plot_learning_curves(self, hist):
        acc      = hist.history["accuracy"]
        val_acc  = hist.history["val_accuracy"]
        loss     = hist.history["loss"]
        val_loss = hist.history["val_loss"]
        epochs = range(1, len(acc)+1)
        fig, (ax1, ax2) = plt.subplots(1,2, figsize=(12,5))
        ax1.plot(epochs, acc, label="train")
        ax1.plot(epochs, val_acc, label="val")
        ax1.set_title("Accuracy"); ax1.legend(); ax1.grid(True)
        ax2.plot(epochs, loss, label="train")
        ax2.plot(epochs, val_loss, label="val")
        ax2.set_title("Loss"); ax2.legend(); ax2.grid(True)
        plt.tight_layout(); plt.show()

    def _plot_confusion(self, y_true, y_prob, thr=0.5):
        y_pred = (y_prob >= thr).astype(int)
        cm = confusion_matrix(y_true, y_pred)
        sns.heatmap(cm, annot=True, fmt="d", cmap="Blues",
                    xticklabels=["Benign", "Malignant"],
                    yticklabels=["Benign", "Malignant"])
        plt.xlabel("Predicted"); plt.ylabel("True"); plt.title("Confusion matrix"); plt.show()

    def _plot_roc(self, y_true, y_prob):
        fpr, tpr, _ = roc_curve(y_true, y_prob)
        plt.figure(figsize=(5,4))
        plt.plot(fpr, tpr, lw=2)
        plt.plot([0,1], [0,1], "--")
        plt.xlabel("FPR"); plt.ylabel("TPR"); plt.title("ROC curve"); plt.show()
