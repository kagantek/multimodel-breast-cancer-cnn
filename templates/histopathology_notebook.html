class DataModule:
    def __init__(self, cfg: Config):
        self.cfg = cfg
        random.seed(cfg.SEED)
        np.random.seed(cfg.SEED)

    def prepare_splits(self):
        # 1 list & (optionally) down-sample
        neg = tf.io.gfile.glob(str(self.cfg.ROOT / "*" / "0" / "*.png"))
        pos = tf.io.gfile.glob(str(self.cfg.ROOT / "*" / "1" / "*.png"))
        if self.cfg.LIMIT is not None:
            n_pos = int(self.cfg.LIMIT * self.cfg.POS_RATIO)
            n_neg = self.cfg.LIMIT - n_pos
            pos = random.sample(pos, min(n_pos, len(pos)))
            neg = random.sample(neg, min(n_neg, len(neg)))
        paths = pos + neg
        labels = [1]*len(pos) + [0]*len(neg)
        tmp = list(zip(paths, labels))
        random.shuffle(tmp)
        self.paths, self.labels = zip(*tmp)

        # 2 train/val/test split
        x_tmp, self.x_test, y_tmp, self.y_test = train_test_split(
            self.paths, self.labels,
            test_size=0.15, stratify=self.labels,
            random_state=self.cfg.SEED
        )
        val_frac = 0.15/0.85
        (self.x_train, self.x_val,
         self.y_train, self.y_val) = train_test_split(
            x_tmp, y_tmp,
            test_size=val_frac, stratify=y_tmp,
            random_state=self.cfg.SEED
        )

    def load_arrays(self):
        def read_resize(p):
            im = cv2.imread(p, cv2.IMREAD_COLOR)
            im = cv2.cvtColor(im, cv2.COLOR_BGR2RGB)
            return cv2.resize(im, self.cfg.IMG_SIZE, interpolation=cv2.INTER_AREA)

        def block(plist):
            arr = np.empty((len(plist), *self.cfg.IMG_SIZE, 3), dtype=np.uint8)
            for i,p in enumerate(plist):
                arr[i] = read_resize(p)
            return arr

        self.x_train = block(self.x_train); gc.collect()
        self.x_val   = block(self.x_val)
        self.x_test  = block(self.x_test)
        self.y_train = np.array(self.y_train, dtype=np.uint8)
        self.y_val   = np.array(self.y_val,   dtype=np.uint8)
        self.y_test  = np.array(self.y_test,  dtype=np.uint8)

    def make_generators(self):
        aug = ImageDataGenerator(**self.cfg.AUG_KWARGS)
        plain = ImageDataGenerator()
        self.train_gen = aug.flow(self.x_train, self.y_train,
                                  batch_size=self.cfg.BATCH,
                                  shuffle=True, seed=self.cfg.SEED)
        self.val_gen   = plain.flow(self.x_val,   self.y_val,
                                    batch_size=self.cfg.BATCH,
                                    shuffle=False)
        self.test_gen  = plain.flow(self.x_test,  self.y_test,
                                    batch_size=self.cfg.BATCH,
                                    shuffle=False)
        # free memory
        del self.x_train, self.x_val, self.x_test
        gc.collect()

class ModelFactory:
    @staticmethod
    def _build(backbone_fn, preprocess_fn, cfg: Config, unfreeze_prefix):
        inp = Input(shape=(*cfg.IMG_SIZE,3), name="input_rgb")
        x = Lambda(preprocess_fn, name="preproc")(inp)
        base = backbone_fn(include_top=False, weights="imagenet", input_tensor=x)
        for l in base.layers: l.trainable = False
        x = GlobalAveragePooling2D()(base.output)
        x = Dropout(0.5)(x)
        x = Dense(256, activation="relu")(x)
        x = Dropout(0.25)(x)
        out = Dense(1, activation="sigmoid", name="pred")(x)
        m = Model(inp, out)
        m._unfreeze_prefix = unfreeze_prefix  # type: ignore
        m.compile(
            optimizer=tf.keras.optimizers.Adam(1e-4),
            loss="binary_crossentropy",
            metrics=["accuracy",
                     tf.keras.metrics.AUC(name="auc"),
                     tf.keras.metrics.Precision(name="precision"),
                     tf.keras.metrics.Recall(name="recall")]
        )
        return m

    @classmethod
    def resnet50(cls, cfg): return cls._build(resnet50.ResNet50, resnet50.preprocess_input, cfg, ["conv4","conv5"])
    @classmethod
    def vgg16(cls, cfg):   return cls._build(vgg16.VGG16,       vgg16.preprocess_input,   cfg, ["block5"])
    @classmethod
    def densenet121(cls,cfg): return cls._build(densenet.DenseNet121, densenet.preprocess_input, cfg, ["conv4","conv5"])

class Trainer:
    def __init__(self, cfg: Config): self.cfg=cfg

    def fine_tune(self, model, train_gen, val_gen):
        # stage1
        hist1 = model.fit(
            train_gen, validation_data=val_gen,
            epochs=5, verbose=2
        )
        # unfreeze tail
        for l in model.layers:
            if any(l.name.startswith(pref) for pref in model._unfreeze_prefix):
                l.trainable = True
        model.compile(
            optimizer=tf.keras.optimizers.Adam(1e-5),
            loss="binary_crossentropy",
            metrics=model.metrics
        )
        hist2 = model.fit(
            train_gen, validation_data=val_gen,
            epochs=25, verbose=2
        )
        # merge
        for k in hist1.history: hist1.history[k] += hist2.history[k]
        return hist1

class Evaluator:
    def __init__(self, cfg: Config):
        self.cfg = cfg

    def predict_dataset(self, model, ds):
        y_true = np.concatenate([y.numpy() for _, y in ds], axis=0)
        y_prob = model.predict(ds, verbose=0).ravel()
        return y_true, y_prob

    def metrics_dict(self, y_true, y_prob, thr=0.5):
        y_pred = (y_prob >= thr).astype(int)
        return dict(
            accuracy=accuracy_score(y_true, y_pred),
            precision=precision_score(y_true, y_pred, zero_division=0),
            recall=recall_score(y_true, y_pred, zero_division=0),
            f1=f1_score(y_true, y_pred, zero_division=0),
            auc=roc_auc_score(y_true, y_prob),
        )
    
    def search_threshold(self, y_true, y_prob):
        best = {"tau":None, "f1":0, "prec":0, "rec":0}
        for t in np.linspace(0.01,0.99,99):
            y_hat = (y_prob>=t).astype(int)
            p,r,f,_ = precision_score(y_true,y_hat), recall_score(y_true,y_hat), f1_score(y_true,y_hat), None
            if f>best['f1']:
                best.update({"tau":t, "f1":f, "prec":p, "rec":r})
        return best

    def full_report(self, model, test_ds, history):
        y_true, y_prob = self.predict_dataset(model, test_ds)
        stats = self.metrics_dict(y_true, y_prob, thr=0.5)
        print("\nTest metrics:")
        for k, v in stats.items():
            print(f"{k:10s}: {v:.4f}")
        print("\nDetailed report:\n", classification_report(y_true, (y_prob>=0.5).astype(int), digits=4))
        self._plot_learning_curves(history)
        self._plot_confusion(y_true, y_prob)
        self._plot_roc(y_true, y_prob)

    def _plot_learning_curves(self, hist):
        acc      = hist.history["accuracy"]
        val_acc  = hist.history["val_accuracy"]
        loss     = hist.history["loss"]
        val_loss = hist.history["val_loss"]
        epochs = range(1, len(acc)+1)
        fig, (ax1, ax2) = plt.subplots(1,2, figsize=(12,5))
        ax1.plot(epochs, acc, label="train")
        ax1.plot(epochs, val_acc, label="val")
        ax1.set_title("Accuracy"); ax1.legend(); ax1.grid(True)
        ax2.plot(epochs, loss, label="train")
        ax2.plot(epochs, val_loss, label="val")
        ax2.set_title("Loss"); ax2.legend(); ax2.grid(True)
        plt.tight_layout(); plt.show()

    def _plot_confusion(self, y_true, y_prob, thr=0.5):
        y_pred = (y_prob >= thr).astype(int)
        cm = confusion_matrix(y_true, y_pred)
        sns.heatmap(cm, annot=True, fmt="d", cmap="Blues",
                    xticklabels=["Benign", "Malignant"],
                    yticklabels=["Benign", "Malignant"])
        plt.xlabel("Predicted"); plt.ylabel("True"); plt.title("Confusion matrix"); plt.show()

    def _plot_roc(self, y_true, y_prob):
        fpr, tpr, _ = roc_curve(y_true, y_prob)
        plt.figure(figsize=(5,4))
        plt.plot(fpr, tpr, lw=2)
        plt.plot([0,1], [0,1], "--")
        plt.xlabel("FPR"); plt.ylabel("TPR"); plt.title("ROC curve"); plt.show()
